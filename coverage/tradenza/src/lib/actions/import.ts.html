// ─── Parse Tradezella row ─────────────────────────────────────────────────────
 
function parseDirection(value: string): 'long' | 'short' {
  const v = value.toLowerCase().trim()
  if (['buy', 'long', 'b', 'l'].includes(v)) return 'long'
  if (['sell', 'short', 's'].includes(v)) return 'short'
  return 'long' // fallback
}
 
function stripTzAbbrev(value: string): string {
  return value
    .trim()
    .replace(/\s+[A-Za-z]{2,5}$/, '')
    .trim()
}
 
function parseDate(value: string): Date | null {
  if (!value || value.trim() === '') return null
  const cleaned = stripTzAbbrev(value)
  const iso = /^\d{4}-\d{2}-\d{2}\s+\d/.test(cleaned) ? cleaned.replace(' ', 'T') : cleaned
  const d = new Date(iso)
  if (!isNaN(d.getTime())) return d
  const fallback = new Date(cleaned)
  return isNaN(fallback.getTime()) ? null : fallback
}
 
function findHeader(headers: string[], candidates: string[]): string | null {
  const norm = headers.map((h) => h.trim().toLowerCase())
  for (const c of candidates) {
    const i = norm.indexOf(c)
    if (i !== -1) return headers[i]
  }
  return null
}
 
function resolveDate(
  row: Record<string, string>,
  dateCol: string | null,
  timeCol: string | null,
  datetimeCol: string | null,
): Date | null {
  if (dateCol && row[dateCol]?.trim()) {
    const date = row[dateCol].trim()
    const time = timeCol ? stripTzAbbrev(row[timeCol] ?? '') : ''
    return parseDate(time ? `${date} ${time}` : date)
  }
  if (datetimeCol && row[datetimeCol]?.trim()) return parseDate(row[datetimeCol])
  if (timeCol && row[timeCol]?.trim()) return parseDate(row[timeCol])
  return null
}
 
function parseNumber(value: string): number | null {
  if (!value || value.trim() === '') return null
  const cleaned = value.replace(/[$,\s]/g, '')
  const n = parseFloat(cleaned)
  return isNaN(n) ? null : n
}
 
// ─── Import from Tradezella CSV ──────────────────────────────────────────────────
 
export async function importFromTradezella(rows: Record<string, string>[], filename: string): Promise<ImportResult> {
  const { userId } = await auth()
  if (!userId) throw new Error('Unauthorized')
 
  const errors: string[] = []
  let imported = 0
  let skipped = 0
 
  const existingTrades = await db.query.trades.findMany({
    where: and(eq(trades.userId, userId), eq(trades.importSource as any, 'tradezella')),
    columns: { externalId: true },
  })
  const existingIds = new Set(existingTrades.map((t) => t.externalId).filter(Boolean))
 
  const headers = Object.keys(rows[0] ?? {})
  const mapping = detectColumns(headers)
  const valueOf = (row: Record<string, string>, field: string) => (mapping[field] ? (row[mapping[field]] ?? '') : '')
 
  const entryDateCol = findHeader(headers, ['open date', 'entry date'])
  const entryTimeCol = findHeader(headers, ['open time', 'entry time'])
  const entryDtCol = findHeader(headers, ['opened', 'date/time', 'datetime', 'entry datetime'])
  const exitDateCol = findHeader(headers, ['close date', 'exit date'])
  const exitTimeCol = findHeader(headers, ['close time', 'exit time'])
  const exitDtCol = findHeader(headers, ['closed', 'exit datetime'])
 
  const toInsert = []
 
  for (let i = 0; i < rows.length; i++) {
    const row = rows[i]
    const rowNum = i + 2
    const get = (field: string) => valueOf(row, field)
 
    try {
      if (!get('symbol')?.trim()) {
        errors.push(t('validation.import.missingSymbol', { row: rowNum }))
        skipped++
        continue
      }
      if (!get('entryPrice') || parseNumber(get('entryPrice')) === null) {
        errors.push(t('validation.import.invalidEntryPrice', { row: rowNum }))
        skipped++
        continue
      }
 
      const entryDatetime = resolveDate(row, entryDateCol, entryTimeCol, entryDtCol)
      if (!entryDatetime) {
        errors.push(t('validation.import.invalidEntryDate', { row: rowNum }))
        skipped++
        continue
      }
 
      const direction = parseDirection(get('direction') || 'long')
 
      // Deduplication key: symbol + entry time + direction
      const externalId = `${get('symbol').trim()}_${entryDatetime.toISOString()}_${direction}`
      if (existingIds.has(externalId)) {
        skipped++
        continue
      }
 
      const entryPrice = parseNumber(get('entryPrice'))!
      const exitPrice = parseNumber(get('exitPrice'))
      const entryQuantity = parseNumber(get('quantity')) ?? 1
      const fees = parseNumber(get('fees')) ?? 0
 
      // P&L
      let grossPnl: string | null = get('grossPnl')
        ? ((parseNumber(get('grossPnl')) ?? null)?.toString() ?? null)
        : null
      let netPnl: string | null = get('netPnl') ? ((parseNumber(get('netPnl')) ?? null)?.toString() ?? null) : null
 
      if (!netPnl && exitPrice !== null) {
        const pnl = calculatePnl(direction, entryPrice, exitPrice, entryQuantity, fees)
        grossPnl = pnl.grossPnl.toString()
        netPnl = pnl.netPnl.toString()
      }
 
      const symbolUpper = get('symbol').trim().toUpperCase()
      const mult = contractMultiplier(symbolUpper)
      const isFutures = mult > 0
 
      toInsert.push({
        userId,
        symbol: symbolUpper,
        direction,
        status: (exitPrice !== null ? 'closed' : 'open') as 'closed' | 'open',
        assetClass: (isFutures ? 'futures' : 'stocks') as 'futures' | 'stocks',
        entryPrice: entryPrice.toString(),
        entryQuantity: entryQuantity.toString(),
        entryDatetime,
        exitPrice: exitPrice?.toString() ?? null,
        exitQuantity: entryQuantity.toString(),
        exitDatetime: resolveDate(row, exitDateCol, exitTimeCol, exitDtCol),
        fees: fees.toString(),
        grossPnl,
        netPnl,
        setupName: get('setupName')?.trim() || null,
        notes: get('notes')?.trim() || null,
        importSource: 'tradezella',
        externalId,
        extra: isFutures ? { contractMultiplier: mult } : undefined,
      })
 
      existingIds.add(externalId)
    } catch (err) {
      errors.push(
        t('validation.import.unknownError', {
          row: rowNum,
          error: err instanceof Error ? err.message : t('validation.import.unknownErrorShort'),
        }),
      )
      skipped++
    }
  }
 
  // Bulk insert in batches of 100
  if (toInsert.length > 0) {
    const chunkSize = 100
    for (let i = 0; i < toInsert.length; i += chunkSize) {
      const chunk = toInsert.slice(i, i + chunkSize)
      await db.insert(trades).values(chunk)
    }
    imported = toInsert.length
  }
 
  // Import log
  await db.insert(importLogs).values({
    userId,
    filename,
    source: 'tradezella',
    totalRows: rows.length,
    importedRows: imported,
    skippedRows: skipped,
    errorRows: errors.length,
    errors: errors.length > 0 ? errors : null,
  })
 
  revalidatePath('/dashboard')
  revalidatePath('/trades')
 
  return { total: rows.length, imported, skipped, errors }
}
 
// ─── Export to CSV ────────────────────────────────────────────────────────────
 
export async function exportTradesToCsv(ids?: string[]): Promise<string> {
  const { userId } = await auth()
  if (!userId) throw new Error('Unauthorized')
 
  const rows = await db.query.trades.findMany({
    where: ids && ids.length > 0 ? and(eq(trades.userId, userId), inArray(trades.id, ids)) : eq(trades.userId, userId),
    orderBy: (t, { asc }) => [asc(t.entryDatetime)],
  })
 
  const headers = [
    'Symbol',
    'Side',
    'Qty',
    'Entry Price',
    'Exit Price',
    'Entry Time',
    'Exit Time',
    'Gross P&L',
    'Net P&L',
    'Commission',
    'Setup',
    'Notes',
    'Rating',
  ]
 
  const escape = (v: string | null | undefined) => {
    if (v === null || v === undefined) return ''
    const s = String(v)
    if (s.includes(',') || s.includes('"') || s.includes('\n')) {
      return `"${s.replace(/"/g, '""')}"`
    }
    return s
  }
 
  const csvRows = rows.map((t) =>
    [
      escape(t.symbol),
      escape(t.direction === 'long' ? 'Buy' : 'Sell'),
      escape(t.entryQuantity),
      escape(t.entryPrice),
      escape(t.exitPrice),
      escape(t.entryDatetime.toISOString()),
      escape(t.exitDatetime?.toISOString()),
      escape(t.grossPnl),
      escape(t.netPnl),
      escape(t.fees),
      escape(t.setupName),
      escape(t.notes),
      escape(t.rating?.toString()),
    ].join(','),
  )
 
  return [headers.join(','), ...csvRows].join('\n')
}
